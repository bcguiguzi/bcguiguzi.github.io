将一个毫秒数值分别转换为天，时，分，秒

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
  </head>
  <body></body>
  <script>
    var d = new Date();
    let a1970 = d.valueOf();
    //下面是将一个毫秒数值分别转换为天，时，分，秒
    a1970 = a1970 / 1000; //除以1000，将毫秒数转成秒数（1秒=1000毫秒）
    let sec = parseInt(a1970 % 60); //余60，得到秒数（1分钟=60秒）
    let min = parseInt(a1970 / 60) % 60; //除以60，余60，得到分钟数（1小时=60分钟）
    let hour = parseInt(a1970 / 3600) % 24; //除以3600，余24，得到小时数（1天=24小时）
    let day = parseInt(a1970 / (3600 * 24)); //除以86400，得到天数
    document.write(
      "距离1970年1月1日午夜以来过去了: " +
        day +
        "天 " +
        hour +
        "时" +
        min +
        "分" +
        sec +
        "秒 "
    );
  </script>
</html>
