```python
def equal_word(B,A): # A ->关键词   B->品牌词
    A,B=str(A),str(B)

    if B in A :  
        return False  #如果包含，返回False
    else:
        return True  #如果不包含，则返回True
df2=df1[['S_key_word','S_brand']]
df1['bool']=df2.apply(lambda x :equal_word(x['S_brand'] ,x['S_key_word']),axis=1)
