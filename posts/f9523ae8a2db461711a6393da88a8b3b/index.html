easyexcel字体加粗

   public static void main(String[] args) {
        List<Data> dataList = new ArrayList<>();
        dataList.add(new Data("Data 1"));
        dataList.add(new Data("Data 2"));
        dataList.add(new Data("Data 3"));

        // 设置加粗字体
        WriteCellStyle boldCellStyle = new WriteCellStyle();
        WriteFont boldFont = new WriteFont();
        boldFont.setBold(true);
        boldCellStyle.setWriteFont(boldFont);

        // 样式策略
        HorizontalCellStyleStrategy styleStrategy = new HorizontalCellStyleStrategy(boldCellStyle, boldCellStyle);

        // 写入 Excel 文件
        EasyExcel.write("BoldCellDemo.xlsx", Data.class)
                .registerWriteHandler(styleStrategy)
                .sheet("Sheet1")
                .doWrite(dataList);
    }

    public static class Data {
        @ExcelProperty("Data")
        private String data;

        public Data(String data) {
            this.data = data;
        }

        public String getData() {
            return data;
        }

        public void setData(String data) {
            this.data = data;
        }
    }
