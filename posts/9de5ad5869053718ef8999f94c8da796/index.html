#include <iostream>
#include <cmath>

using namespace std;

int main()
{
	int N = 0;  // 同学数 
	int i = 0;
	int j = 0;
	int k = 0;
	int flag = 0;
	int a[1001][5] = {0};  // 将成绩综合 放到 第 4 列 

	cin >> N;
	
	for (i=0; i<N; i++)  //输入 
	{
		for (j=0; j<3; j++)
		{
			cin >> a[i][j];
			a[i][4] += a[i][j];
		}
	}	
	int cnt = 0;	
	for (i=0; i<N; i++)	//判断
	{	 
		for (k=i+1; k<N; k++)
		{	
			flag = 0;	
			for (j=0; j<3; j++)
			{
				if (abs(a[i][j]-a[k][j]) > 5)  
				{
					break;	
				}
				flag++;
			}
			if (flag==3 && abs(a[i][4]-a[k][4])<=10)
			{
					cnt++;
			}
		}
	} 
	
	cout << cnt; 
	
	return 0;
}
