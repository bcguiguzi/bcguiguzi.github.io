Easy h-index（水题）

题目链接：http://exam.upc.edu.cn/problem.php?id=7034

思路：从大到小枚举h，主要是理解题目，要求至少有h篇   引文不低于h   的论文

真的坑，我一直都在错，错得我都怀疑人生了。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll a[250005];
int main()
{
    ll n;
    while(scanf("%lld",&n)!=EOF){
        ll sum=0;
        for(int i=0;i<=n;i++){
            scanf("%lld",&a[i]);
        }
        for(int i=n;i>=0;i--){
            sum+=a[i];
            if(sum>=i){
                printf("%d\n",i);break;
            }
        }
    }
    return 0;
}
