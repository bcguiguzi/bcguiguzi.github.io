c++:遍历vector容器的5种方法

1.迭代器
2.C++11 新增关键字auto
3.数组遍历法
4.基于范围的for循环
5.利用成员函数 at()进行访问
注意：[]与at()既可以取值也可以赋值，但是建议用at()，因为超出索引会报错，但是要牺牲性能的损耗

#include <vector>
#include <iostream>
using namespace std;
 
int main()
{
    vector<int>obj;
    
    for(int i=0;i<10;i++)
    {
        obj.push_back(i);
        cout<<obj[i]<<",";    
    }
    cout << endl << "vector size is :" << obj.size() << endl;
    
    //iterator遍历
    for(vector<int>::iterator it = obj.begin(); it != obj.end(); it++)
    {
        cout << *it << ",";
    }
    cout << endl;
    
    //数组循环遍历
    for(int i=0;i<obj.size();i++)
    {
        cout<<obj[i]<<",";
    }
    cout << endl;
     //基于at()成员函数的遍历
    for(int i=0;i<obj.size();i++)
    {
        cout<<obj.at(i)<<",";
    }
    cout << endl;
    
     //auto关键字遍历遍历
    for(auto it = obj.begin(); it != obj.end(); it++)
    {
        cout << *it << ",";
    }
    cout << endl;
 
 //基于范围的for循环
    for (auto iter : obj) 
    {
        cout << iter << ",";
    }
    cout << endl;
 
    return 0;
}
