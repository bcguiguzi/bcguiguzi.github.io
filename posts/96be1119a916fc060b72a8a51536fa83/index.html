顺序表的查找----按位查找与按值查找

  1. 按位查找
#define  _CRT_SECURE_NO_WARNINGS 1
#include<stdio.h>
#include<stdlib.h>
#define InitSize 10
typedef struct {
	int* data;
	int MaxSize;
	int length;
}SqList;
//初始化表
void InitList(SqList& L) {
	L.data = (int*)malloc(InitSize * sizeof(int));
	L.length = 0;
	L.MaxSize = InitSize;
}
//按位查找
int GetElem(SqList& L, int i) {
	return L.data[i - 1];
}
int main() {
	SqList L;
	InitList(L);
	//插入元素
	for (int i = 0; i < 10; i++) {
		L.data[i] = i;
		L.length++;
	}
	//打印元素
	for (int i = 0; i < 10; i++) {
		printf("data[%d]=%d\n", i, L.data[i]);
	}
	//记录返回结果
	int result = GetElem(L, 3);
	printf("位序为3的元素为：%d\n", result);
	return 0;
}
  1. 按值查找
#define  _CRT_SECURE_NO_WARNINGS 1
#include<stdio.h>
#include<stdlib.h>
#define InitSize 10
typedef struct {
	int* data;
	int MaxSize;
	int length;
}SqList;
//初始化表
void InitList(SqList& L) {
	L.data = (int*)malloc(InitSize * sizeof(int));
	L.length = 0;
	L.MaxSize = InitSize;
}
//按值查找
int LocateElem(SqList& L, int e) {
	for (int i = 0; i < L.length; i++) {
		if (L.data[i] == e) {
			return i + 1;//返回位序
		}
	}
	return 0;
}
int main() {
	SqList L;
	InitList(L);
	//插入元素
	for (int i = 0; i < 10; i++) {
		L.data[i] = i;
		L.length++;
	}
	//打印元素
	for (int i = 0; i < 10; i++) {
		printf("data[%d]=%d\n", i, L.data[i]);
	}
	//记录返回结果
	int result = LocateElem(L, 5);//查找元素5
	printf("查找成功，位序为：%d\n", result);
	return 0;
}
