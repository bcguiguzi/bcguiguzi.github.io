前台ajax:

$.ajax({
	url : "<%=request.getContextPath()%>/xxx", // 请求地址
	type : "get", // 请求的类型
	data : { person_id : person_id }, // 请求的数据,规定连同请求发送到服务器的数据
	success : function(data){ // 请求成功时的回调函数
		flag = rs.success;
        	if(flag){
                alert("删除成功!");
            }
	},
})
