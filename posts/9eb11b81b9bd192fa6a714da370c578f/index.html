select top 200 A.*,B.num from dbo.ResultAll  A left join (
 select top 1000 名称,COUNT(1) num from   dbo.ResultAll where 入选日期='20240314'  group by 名称 order by COUNT(1) desc) B ON   A.名称=B.名称
 where  A.入选日期='20240314' and 涨幅>9 and B.num>=2 and 主力净额>0  and DDE散户数量<0

 and 现价> MA120 and 现价> MA250 and MA120>MA250
