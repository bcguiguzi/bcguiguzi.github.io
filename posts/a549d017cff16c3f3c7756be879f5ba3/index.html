javascript打印交替换色99乘法表

<script>
	var table='<table border ="1">';
	//99乘法表
	for(var r=1;r<10;r++){
		//奇数行
	   if(r%2==1){var tr='<tr class ="odd_color">';
      //偶数行
	}else{var tr='<tr class ="even_color">';

	}
		
		for(var c=1;c<=r;c++){
			tr +='<td>'+c+'*'+r+'='+r*c+'</td>';

		}
		tr+='<tr>';
		//拼接到table
		table+=tr
		}
		//把table的闭合标签拼接
		var tab = table+'<table>'
		document.write(tab)
</script>
