不用sort 将一个列表从小到大排列

list=[12,53,5,24,98,62]
new_list=[]  #定义一个空列表
def get_min(list):
    #获取列表最小值
    a=min(list)
    #原列表删除最小值
    list.remove(a)
    #将最小值加入新列表
    new_list.append(a)
    #保证列报中有值,递归调用获取最小值,直到所有制获取完,并加入新列表返回新列表
    if len(list)>0:
        get_min(list)
    return new_list
new_list=get_min(list)
print(new_list)
