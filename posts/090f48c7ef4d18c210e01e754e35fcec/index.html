乌班图静态指定IP

This is the network config written by ‘subiquity’

sudo vi /etc/netplan/00-installer-config.yaml    ##进入配置文件

network:
  ethernets:
    enp1s0:
      dhcp4: false
      addresses:
        - 192.168.1.11/24
      routes:
        - to: default
          via: 12.168.1.1
      nameservers:
          addresses:
            - 192.168.1.1
            - 114.114.114.114
      dhcp6: false
      match:
        macaddress: 8c:dc:d4:6c:1b:2f
      addresses:
              - fe80::8edc:d4ff:fe6c:1b2f/64
  version: 2
~                                                                               
~                                                                               

sudo netplan apply   ##重启网卡
