单词翻转

class Solution {//先整体反转，然后在单个单词翻转
public:
    void revers(string &str,int i,int j)//要采用引用传递
{
	while (i < j)
	{
		char tmp = str[j];
		str[j] = str[i];
		str[i] = tmp;
		i++;
		j--;
	}
}
string ReverseSentence(string str) {
	if (str.size() == 0)
		return "";
	int len = str.size();
	revers(str,0,len-1);
	//string s = str;
	int i = 0, j = 0;
	while (str[i]!='\0')
	{
		if (str[i] == ' ')
		{
			i++;
			j++;
			continue;
		}
		else if (str[j] == ' ' || str[j] == '\0')
		{
			j--;
			revers(str, i, j);
			j++;
			i = j;
		}
		else
			j++;
	}
	return str;
}
   
};
