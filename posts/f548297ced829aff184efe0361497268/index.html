CCF：201912-1 报数（C语言）

#include <stdio.h>
int main()
{
	int i,n,asum=0,bsum=0,csum=0,dsum=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		if(i%4==1&&(i%7==0||i/100==7||i%10==7||i/10%10==7))
		{
			asum++;
			n++;
		}
	    if(i%4==2&&(i%7==0||i/100==7||i%10==7||i/10%10==7))
		{
			bsum++;
			n++;
		}
		if(i%4==3&&(i%7==0||i/100==7||i%10==7||i/10%10==7))
		{
			csum++;
			n++;
		}
		if(i%4==0&&(i%7==0||i/100==7||i%10==7||i/10%10==7))
		{
			dsum++;
			n++;
		}
	}
	printf("%d\n",asum);
	printf("%d\n",bsum);
	printf("%d\n",csum);
	printf("%d\n",dsum);
	return 0;
}
