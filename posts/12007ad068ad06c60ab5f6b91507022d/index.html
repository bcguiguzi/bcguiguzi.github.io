初

很多时候，我们会遇到需要使用 字符串 转 数字 的功能，我们可以用 strconv.Atoi() 来进行转换
直接上代码：

代码：

/*
@Time : 2021/6/23 13:40
@Author : dao
@File : 字符串转整型
@Software: GoLand
*/

package main

import (
	"fmt"
	"strconv"
)

func main() {
	//声明：并赋值：一个字符串
	a := "10"
	aint, _ := strconv.Atoi(a)
	fmt.Println(aint) //10
	fmt.Println(fmt.Sprintf("aint+1=%d", aint+1)) //aint+1=11
}
