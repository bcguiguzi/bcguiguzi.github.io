3) 添加数据，防止SQL注入攻击
在添加到查询语句中之前，query方法会自动把参数转义，以防遭受到SQL注入攻击

exports.add = function(db, req, res) {
    exports.parseReceivedData(req, function(work) {
    var querySqlAdd = "INSERT INTO work (hours, date, description)"
                + " VALUES (?, ?, ?)";

            querySqlAdd,

            function(err) {

                }
                exports.show(db, res);
            }
        );
    });
};
