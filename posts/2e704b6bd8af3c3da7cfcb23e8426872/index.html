数据库常用的四种方法

排序查询
select 列表属性，列表属性 from 列表名order by 列表属性，列表属性;
删除
delete from 列表名 where 列表属性=“值”
插入
insert into 列表名 values(值，值，值，值）
insert into 列表名 （列1，列2，…）valves(值1，值2…)
修改
update 列表名 set 列表属性=新值 where 列表属性=某值
