js如下:

  function submit() {
        var project_status = document.getElementById("showType").value;
        projectQuery(project_status);
    }

    function projectQuery(project_status) {
        $('#telrecord-grid').bootstrapTable(
                "refresh",
                {
                    url:"/project/query?project_status="+project_status
                }
        );
    }
