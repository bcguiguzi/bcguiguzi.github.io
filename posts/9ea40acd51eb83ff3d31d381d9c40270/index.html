动态规划法解最长公共子序列问题(附带dp数组打印)

import java.util.Arrays;

public class LCS2 {
    private void printDp(int[][] dp){
        for(int i = 0; i < dp.length; i++){
            System.out.println(Arrays.toString(dp[i]));
        }
        System.out.println("--------------------------------");
    }
    public int lcs(char[] arr1, char[] arr2, int m, int n){
        int[][] dp = new int[m+1][n+1];

        for(int i = 0; i <= m; i++){
            for(int j = 0; j <= n; j++){
                if(i == 0 || j == 0){
                    dp[i][j] = 0;
                }
                else if(arr1[i-1] == arr2[j-1]){
                    dp[i][j] = dp[i-1][j-1] + 1;
                }
                else{
                    dp[i][j] = Math.max(dp[i-1][j],dp[i][j-1]);
                }

            }
            printDp(dp);
        }
        return dp[m][n];
    }
    public static void main(String[] args) {
        LCS2 lcs2 = new LCS2();
        String s1 = "ABCBDAB";
        String s2 = "BDCABA";
        char[] arr1 = s1.toCharArray();
        char[] arr2 = s2.toCharArray();
        int m = arr1.length;
        int n = arr2.length;
        System.out.println(lcs2.lcs(arr1, arr2, m, n));
    }
}
