jquery 获取所有的 checkbox 值

<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 4.01 Transitional//EN">
<html>
<head>
    <meta charset="utf-8">
    <script type="text/javascript" src="jquery-1.7.2.min.js"></script>
    <script type="text/javascript">

    function getCheckAdIds() {
        var adIds = "";
        $("input:checkbox[name=check_name]:checked'").each(function(i){
            if(0==i){
                adIds = $(this).val();
            }else{
                adIds += (","+$(this).val());
            }
        });
        alert(adIds);
        return adIds;
    }
    </script>
</head>
<body>

    <input name="check_name" type="checkbox" value="1"/>1
    <input name="check_name" type="checkbox" value="2"/>2
    <input name="check_name" type="checkbox" value="3"/>3
    <input name="check_name" type="checkbox" value="4"/>4
    <input name="check_name" type="checkbox" value="5"/>5
    <input name="check_name" type="checkbox" value="6"/>6<br/>

    <button οnclick="getCheckAdIds()">获取选中结果</button>

</body>
