							http://www.springframework.org/schema/mvc
							http://www.springframework.org/schema/mvc/spring-mvc.xsd">

</beans>

② 扫描包


    <context:component-scan base-package="com.qf.controller"></context:component-scan>

③开启注解

<mvc:annotation-driven></mvc:annotation-driven>

④配置静态资源的访问

<mvc:default-servlet-handler></mvc:default-servlet-handler>
   <!--由于mvc.xml中配置的前端控制器的映射路径为 / 与tomcat默认的静态资源访问路径相同，所以，静态资源路径会失效，所以要配置静态资源访问路径-->

<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:context="http://www.springframework.org/schema/context"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

       http://www.springframework.org/schema/context
       http://www.springframework.org/schema/context/spring-context.xsd
       http://www.springframework.org/schema/tx
       http://www.springframework.org/schema/tx/spring-tx.xsd
       http://www.springframework.org/schema/aop
       http://www.springframework.org/schema/aop/spring-aop.xsd
        ">
    <context:property-placeholder location="classpath:jdbc.properties"/>

<!--这里使用的阿里的druid连接池-->
    <bean id="dataSource" class="com.alibaba.druid.pool.DruidDataSource">
        <property name="driverClassName" value="${jdbc.driver}"/>
        <property name="url" value="${jdbc.url}"/>
        <property name="username" value="${jdbc.username}"/>
        <property name="password" value="${jdbc.password}"/>
    </bean>

<!--创建sqlSessionFactory工厂-->
    <bean id="sqlSessionFactory" class="org.mybatis.spring.SqlSessionFactoryBean">
        <property name="dataSource" ref="dataSource"/>
        <property name="mapperLocations">
            <list>
                <value>classpath:com/qf/dao/UserDaoMapper.xml</value>
            </list>
        </property>
        <!--为实体类起别名，在pom.xml中要添加build，（代码在下一段）-->
        <property name="typeAliasesPackage" value="com.qf.entity"/>
        
<!--分页插件-->
        <property name="plugins">
            <bean class="com.github.pagehelper.PageInterceptor"></bean>
        </property>
    </bean>

<!--扫描dao层，自动生成对应的bean-->
    <bean class="org.mybatis.spring.mapper.MapperScannerConfigurer">
        <property name="basePackage" value="com.qf.dao"/>
        <property name="sqlSessionFactoryBeanName" value="sqlSessionFactory"/>
    </bean>
</beans>

2.service的配置
创建UserService的接口与实现类
实现类中要有@Service 和userDao自动注入

@Service
public class UserServiceImpl implements UserService {
    @Autowired
    private UserDao userDao;
    
    @Override
    @Transactional(propagation = Propagation.SUPPORTS)
    public List<User> selectAll() {
        return userDao.selectAll();
    }
    ｝

在spring-context.xml要加入注解扫描

 <!--mvc.xml中扫描了controller包，spring-context.xml中在扫描的话可能会造成接口污染，所以在这里要声明不扫描controller包-->
<context:component-scan base-package="com.qf">
        <context:exclude-filter type="annotation" expression="org.springframework.stereotype.Controller"/>
    </context:component-scan>

加入事务管理器，管理和sqlSessionFactory的同一个数据源
注解的事务配置


<!--事务-->
    <bean id="txManger" class="org.springframework.jdbc.datasource.DataSourceTransactionManager">
        <property name="dataSource" ref="dataSource"/>
    </bean>
    <tx:annotation-driven transaction-manager="txManger"/>
</beans>

在userServiceImpl中配置事务注解：
@Transactional(propagation = Propagation.SUPPORTS)
