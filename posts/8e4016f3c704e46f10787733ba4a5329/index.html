matplotlib隐藏坐标轴

# 法一：
plt.gca().get_xaxis().set_visible(False)
plt.gca().get_yaxis().set_visible(False)
# 法二：
plt.xticks([])
plt.yticks([])
# 法三：
plt.axis('off')
