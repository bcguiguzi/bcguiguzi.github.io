git暂存区（修改待提交区）

提交命令

// 提交指定文件到暂存区（单个/多个）
   $git add 文件名.后缀

// 提交当前文件夹中所有文件到暂存区
   $git add -A
   $git add .

查看命令

// 查看当前分支暂存区的文件
   $git ls-files

撤销命令

// 撤销暂存区指定文件（单个/多个）
   $git reset HEAD -- 文件名.后缀名
   $git rm --cached 文件名.后缀名
   $git rm 文件名.后缀名
