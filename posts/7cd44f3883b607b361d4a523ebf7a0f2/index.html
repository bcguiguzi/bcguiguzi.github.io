1.

查询"01"课程比"02"课程成绩高的学生的信息及课程分数

解：

SELECT s.s_id, s.s_name, b.cid1, b.score1, b.cid2, b.score2
FROM student s
JOIN
(
SELECT a.s_id as id, a.c_id as cid1, a.score as score1, b.c_id as cid2, b.score as score2
FROM stu_sco a
JOIN stu_sco b
ON a.s_id = b.s_id 
WHERE a.score > b.score AND a.c_id = 1 AND b.c_id = 2
) b
ON s.s_id = b.id

结果：

解析：

其实单看下面这个sql一目了然：
这就是上面的b表，实际上这个b表已经能查出相关的信息了，只不过我们需要获取学生的姓名，所以外面还要跟student表join一次。

逻辑其实就是很简单，将成绩表做一个自身连接，由于是要找某一名学生的01课程分数大于02课程分数。因此前提就是必须是同一名学生，因此JOIN的条件为s_id相同。然后判断条件就很简单了。

SELECT a.s_id as id, a.c_id as cid1, a.score as score1, b.c_id as cid2, b.score as score2
FROM stu_sco a
JOIN stu_sco b
ON a.s_id = b.s_id 
WHERE a.score > b.score AND a.c_id = 1 AND b.c_id = 2

2.

查询平均成绩大于等于60分的同学的学生编号和学生姓名和平均成绩

SELECT a.s_name, b.avg_socre
FROM student a
JOIN
(
SELECT s_id, AVG(score) as avg_socre
FROM stu_sco
GROUP BY s_id
HAVING avg_socre > 60
) b
ON a.s_id = b.s_id

解析：

  • 其实就是查询平均成绩大于60的学生 —— b表
  • 然后与a表做关联，查姓名即可

3.

查询所有同学的学号、姓名、选课数、总成绩

SELECT s_name, b.*
FROM student a
JOIN
(
SELECT s_id, COUNT(*) as num, SUM(score) as sum_score
FROM stu_sco
GROUP BY s_id
) b
ON a.s_id = b.s_id

解析：

  • 只看b表，同时使用count(*) 和 SUM(score) 来计算课程数和总分数，最后使用GROUP BY根据id分类
  • 使用a表join是因为需要查询姓名，关联条件则是s_id

5

查询没学过“张三”老师课的同学的学号、姓名

SELECT s_id, s_name
FROM student
WHERE s_id NOT IN 
(
SELECT s_id -- 查询选了张三老师课程的学生 
FROM stu_sco a
JOIN course b  -- 成绩表关联课程表
ON a.c_id = b.c_id
JOIN teacher c  -- 课程表关联教师表
ON b.t_id = c.t_id 
WHERE c.t_name = "张三"
)

思路：

查询学过编号“01”并且也学过编号“02”课程的同学的学号、姓名

SELECT s_id, s_name
FROM student
WHERE s_id IN
(
SELECT DISTINCT(s1.s_id)
FROM stu_sco s1  -- 两张成绩表关联
JOIN stu_sco s2
ON
s1.s_id = s2.s_id -- id相同

SELECT s1.s_id, s1.s_name, s2.c_id, t1.t_name
FROM student s1
JOIN stu_sco s2
ON s1.s_id = s2.s_id  -- 关联学生表查询姓名
JOIN course c1

ON c1.t_id = t1.t_id  -- 关联教师表查姓名

FROM student s

SELECT s1.s_id

9

查询所有课程成绩小于60分的同学的学号、姓名；

思路1：
查询有一门成绩大于60分的所有同学，然后关联学生表not in

注意： 这种方法有一种缺陷，就是如果缺考的学生也会算进来，所以建议用第二种

SELECT s_id, s_name
FROM student
WHERE s_id NOT IN
(
SELECT DISTINCT(s_id)
FROM stu_sco
WHERE score > 60 -- 查询分数大于60的 
)

查询最高成绩都小于60分的同学，关联学生表join

SELECT a.s_id, a.s_name
FROM student a
JOIN 
(
SELECT s_id, MAX(score) max_score
FROM stu_sco
GROUP BY s_id
HAVING max_score < 60 -- 查询最高分数都小于60的
) b
ON a.s_id = b.s_id
