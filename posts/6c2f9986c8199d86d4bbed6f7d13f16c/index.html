//IEnumerable是可枚举类型，一般在迭代时应用广泛，如foreach中要循环访问的集合或数组都实现了IEnumerable接口。只要能够遍历，都直接或间接实现了IEnumerable接口。
static IEnumerable<int> MyWhere(IEnumerable<int> items,Func<int,bool> f)
{
    List<int> result = new List<int>();
    foreach (var item in items)
    {
        if(f(item)==true)
            result.Add(item);
    }

    return result;
}
