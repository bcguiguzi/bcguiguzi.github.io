# 用户输入一个数字，打印每一位数字及其重复的次数


n = "13211445"
dic = {}
for i in range(len(n)):
    print(n[i])
    if n[i] not in dic:
        dic[n[i]] = 1
    else:
        dic[n[i]] += 1
print(dic)
