go || 运算符

使用浮点型参与计算

func main() {
	// 使用浮点型参与计算
	var f1 float32
	f1 = -3.0 / 2
	fmt.Println(f1)

	x := 8
	if x > 5 && x < 10 { //go语言运算符不能连用
		fmt.Println("x在5-10之间")
	}
}

逻辑运算符 && || !

func main() {
	a := 3
	b := 2
	c := 0
	if (a < b) && (a/c > 0) {
		fmt.Println("短路运算")
	}
}

自增、自减

func main() {
	sc := 9
	sc++
	sc--
	fmt.Println(sc)
}
