#输出 1~100 之间不能被 7 整除的数

j = 0    #

for i in range(1,101):    # 遍历1-100取值，定义为变量 i

if i%7 != 0:     # 找出不能被 7 整除的数

print("{:3d}".format(i),end=‘ ‘)    # Format格式化输出

j += 1      # 对输出的 不能被 7 整除的数 进行计数

if j%10 == 0:    # 控制每行输出数字保持10个

print(‘\n‘)   # 每行输出数字到第11个，进行换行操作
