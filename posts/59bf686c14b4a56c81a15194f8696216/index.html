vector<string>怎么用

//两种排序方法

#include<iostream>
#include<string>
#include<vector>
using namespace std;
vector<string>vec;//定义个一个字符串容器

bool lexicographically(){
    int i=0;
    while(i<vec.size()-1){//vec.size() //返回容器中实际数据的个数。
        if(vec[i].compare(vec[i+1])>0){
            return 0;
        }
        i++;
    }
    return 1;
}

bool lengths(){
    int i=0;
    while(i<vec.size()-1){
        if(vec[i].size()>vec[i+1].size())
        {
            return 0;
        }
        i++;
    }
    return 1;
}
int main(){
    int i=0,n;
    string str;
    bool r1,r2;
    cin>>n;
    while(i<n){
        cin>>str;
        vec.push_back(str);//把字符串str压进容器，vec.pop_back();//取出容器中最后一个
        i++;
    }
    r1=lexicographically();
    r2=lengths();
    if(r1&&r2){
        cout<<"both";
    }
    else if(r1&&(!r2)){
        cout<<"lexicographically";
    }
    else if((!r1)&&r2){
        cout<<"lengths";
    }
    else{
        cout<<"none";
    }
    return 0;
}
