【C++】简单说说C++中的多线程

1.代码

#include<iostream>
#include<thread>
#include <mutex>
using namespace std;

mutex mymutex;   //加入锁

void thread1(){
    unique_lock<mutex> lock(mymutex);
    for(int i=0; i<30; i++){
        cout<< "线程1运行" << endl;
    }
};

void thread2(){
    unique_lock<mutex> lock(mymutex);
    for(int i=0; i<30; i++){
        cout<< "线程2运行" << endl;
    }
};

int main() {
    thread t1(thread1);
    thread t2(thread2);
    
    t1.join();
    t2.join();
    
    return 0;
}
