vector的迭代器遍历

vector的迭代器遍历：

#include <iostream>
#include <vector>
using namespace std;
 
// vector容器遍历方式2 —— 迭代器遍历
void traverseVector_2(vector<int> v)
{
	// 注：如果参数为const vector<int> 需要用const_iterator
	vector<int>::iterator it = v.begin();
	// vector<int>::const_iterator iter=v.begin();
	for(; it != v.end(); ++it)
	{
		cout<<(*it)<<" ";
	}
	cout<<endl;
}
