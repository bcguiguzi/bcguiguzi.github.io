#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn = 2e5+5;
const int INF = 1e9+7;
const ll mod =  998244353;
typedef pair<int,int> pii;
ll f_pow(ll a,ll b){
	ll ans = 1;
	while(b){
		if(b&1) ans = (ans*a)%mod;
		a=(a*a)%mod;b>>=1;
	}
	return ans;
}
ll inv_fac[maxn];
ll inv[maxn];
ll fac[maxn];
void inv_table(ll n){
	inv_fac[0]=1;
	inv[1]=1;fac[1]=1;
	for(int i=2;i<=n;i++){
		inv[i] = (mod-mod/i)*inv[mod%i]%mod;
		fac[i] = (fac[i-1]*i)%mod;
	}
	inv_fac[n] = f_pow(fac[n],mod-2);
	for(ll i=n-1;i>=1;i--){
		inv_fac[i] = (inv_fac[i+1] *(i+1)) %mod;
	}
}
ll tree[maxn];
void add(int x,int val){
	while(x<=maxn-5){
		tree[x]=(tree[x]+val)%mod;
		x+=(x&-x);
	}
}
ll query(int x){
	ll ans = 0;
	while(x){
		ans +=tree[x];
		x-=(x&-x);
	}
	return ans;
}
int cnt[maxn];
int main(){
//    freopen("1.txt","r",stdin);
    ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	inv_table(200000);
	int n,m;cin>>n>>m;
	for(int i=1;i<=n;i++){
		int x;cin>>x;
		cnt[x]++;
		add(x,1);
	}
	ll var =fac[n];
	for(int i=1;i<=maxn-5;i++){
		var = var*inv_fac[cnt[i]]%mod;
	}
	ll ans = 0;
	for(int i=1;i<=m;i++){
		int x;cin>>x;
		ll sum = query(x-1);
		ans=(inv[n-i+1]*sum%mod*var%mod+ans)%mod;
		if(cnt[x]==0) break;
		var = var*cnt[x]%mod*inv[n-i+1]%mod;
		cnt[x]--;
		add(x,-1);
		if(i==n&&i<m){
			ans=(ans+1)%mod;
			break;
			//s用完了,t没用完,且s是t的前缀.
		}
	}
	cout<<ans<<"\n";
}
