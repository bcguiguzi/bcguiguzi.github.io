    function show(v) {
        alert(v)

    }

    function show2(v1,v2) {
        alert(v1)
        alert(v2)
    }
    function show3(v) {
            alert(v.name)
    }

    var arr = [
        {
            name: 'wangwu',
            age: 23,
            hight: [180, 180, 180]
        },

        {

   //传递一个参数

    //ht = '<button onclick=show2("' + arr[1].name + '","' + arr[1].age + '")>我是按钮</button>'

    //ht = '<button onclick=show3(' + JSON.stringify(arr[2]) + ')>我是按钮</button>'
