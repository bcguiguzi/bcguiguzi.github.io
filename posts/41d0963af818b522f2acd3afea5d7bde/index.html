递归应用---倒序输出数组

//用递归函数逆序输出数组中的10个元素 。

#include<stdio.h>
void fun(int a[], int i) {
	if (i > 0) {								//不可去掉if后的花括号。否则，fun(a,--i);语句会导致本函数成为死循环 。
		printf_s("%d ", a[i - 1]);
		fun(a, --i);							//注意这里是--i
	}
}
void main() {
	int a[10];
	for (int i = 0; i < 10; i++) {
		a[i] = i;
	}
	printf_s("当前数组为： ");
	for (int i = 0; i < 10; i++) {
		printf_s("%d ", a[i]);
	}
	printf_s("\n倒序输出： ");
	fun(a, 10);
	printf_s("\n");
	system("pause");
}
