- (NSString *)nickNameFirstChar:(NSString *)nickName
{
    char firstChar = ' ';
    if (nickName.length > 0) {
        firstChar = ([nickName cStringUsingEncoding:NSUTF8StringEncoding])[0];
        if (!((firstChar >= '0' && firstChar <= '9') || (firstChar >= 'a' && firstChar <= 'z') || (firstChar >= 'A' && firstChar <= 'Z'))) {
            NSString *pinYin = [NSString stringWithFormat:@"%c",pinyinFirstLetter([nickName characterAtIndex:0])];
            firstChar = ([pinYin cStringUsingEncoding:NSASCIIStringEncoding])[0];
        }
    }
    
    return [[NSString stringWithFormat:@"%c", firstChar] uppercaseString];
}
