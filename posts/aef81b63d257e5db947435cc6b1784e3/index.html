原生javascript 模拟enter事件

模拟enter事件

    let event = document.createEvent('Event');
    event.initEvent('keyup', true, false);
    event = Object.assign(event, {
        ctrlKey: false,
        metaKey: false,
        altKey: false,
        which: 13,
        keyCode: 13,
        key: 'Enter',
        code: 'Enter'
    })
    input.dispatchEvent(event); //let input = document.querySelector("input");
