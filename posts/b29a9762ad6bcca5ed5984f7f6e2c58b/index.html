el-table中cell-class-name的用法（给指定列的单元格设置样式）

<el-table
  :key="selectedRisk.id"
   width="100%"
   ref="table"
   row-key="id"
  :cell-class-name="cellClassName"
   border
   height="calc(100vh - 262px)"
   :data="tableData"
 >
</el-table>
js:
cellClassName (data) {
  if (data.columnIndex === 3) {
    return 'no-padding'
  }
},

css:

/deep/tbody {
  .el-table__cell {
    padding: 0;
  }
  .no-padding {
    .cell {
      padding: 0;
    }
  }
}
