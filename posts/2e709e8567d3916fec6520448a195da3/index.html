自己实现MATLAB支持向量化输入输出的零阶修正贝塞尔函数

function [ y ] = Bessel0( x )
% 此函数用于计算0阶修正贝塞尔函数值
% 支持向量化输入输出
% 作用相当于MATLAB自带的库函数besseli(0,x)
theta =0.001:0.001:2*pi;
length(theta);
t= repmat(x, length(theta), 1);
t= reshape(t,1,[]);
theta= repmat(theta, 1, length(x));
t= t .* cos(theta);
t=reshape(t, [], length(x));
y=0.001/(2*pi) *sum(exp(t), 1);
end
